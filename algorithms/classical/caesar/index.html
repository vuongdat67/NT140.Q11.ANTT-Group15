    def caesar_encrypt(s: str, k: int) -> str:
        k = k % 26
        out = []
        for ch in s:
                if 'A' <= ch <= 'Z':
                        out.append(chr((ord(ch) - 65 + k) % 26 + 65))
                elif 'a' <= ch <= 'z':
                        out.append(chr((ord(ch) - 97 + k) % 26 + 97))
                else:
                        out.append(ch)
        return ''.join(out)

def caesar_decrypt(c: str, k: int) -> str:
        return caesar_encrypt(c, -k)
